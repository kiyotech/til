## 基本
```ruby
class Player
    def walk()
        puts "勇者は荒野を歩いていた"
    end
    
    def attack(enemy)
        puts "勇者は#{enemy}を攻撃した"
    end
end
player1 = Player.new()　#インスタンスを実体化
player1.walk()          #メソッドの呼び出し
player1.attack("スライム")
クラス名.メソッド名(引数)
```

## initializeメソッド
```ruby
class Player
    def initialize(job) # 初期化メソッド
        @job = job　　#　オブジェクトが持つ変数に初期値を設定できる
    end

    def walk()
        puts "#{@job}は、荒野を歩いていた。"
    end
end

player1 = Player.new("戦士")
player1.walk()
```
## attr_accessor
```ruby
class Player
    attr_accessor :job # 外部から読み書きできるようにする
    def initialize(job) 
        @job = job
    end

    def walk()
        puts "#{@job}は、荒野を歩いていた。"
    end
end

player1 = Player.new("戦士")
player1.walk()

player1.job = "勇者"　# jobを勇者に書き換える
player1.walk()
```

## 継承
```ruby
class Greeting
    def initialize()
        @msg = "hello"
        @target = "paiza"
    end
end

class Hello < Greeting #継承している
    # この下に、say_helloメソッドを記述する
    def say_hello()
        puts "#{@msg} #{@target}"
    end
end

player = Hello.new()
player.say_hello()
```
