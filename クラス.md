## 基本
```ruby
class Player
    def walk()
        puts "勇者は荒野を歩いていた"
    end
    
    def attack(enemy)
        puts "勇者は#{enemy}を攻撃した"
    end
end
player1 = Player.new()　#インスタンスを実体化
player1.walk()          #メソッドの呼び出し
player1.attack("スライム")
クラス名.メソッド名(引数)
```

## initializeメソッド
```ruby
class Player
    def initialize(job) # 初期化メソッド
        @job = job　　#　オブジェクトが持つ変数に初期値を設定できる
    end

    def walk()
        puts "#{@job}は、荒野を歩いていた。"
    end
end

player1 = Player.new("戦士")
player1.walk()
```
## attr_accessor
```ruby
class Player
    attr_accessor :job # 外部から読み書きできるようにする
    def initialize(job) 
        @job = job
    end

    def walk()
        puts "#{@job}は、荒野を歩いていた。"
    end
end

player1 = Player.new("戦士")
player1.walk()

player1.job = "勇者"　# jobを勇者に書き換える
player1.walk()
```

## 継承
```ruby
class Greeting
    def initialize()
        @msg = "hello"
        @target = "paiza"
    end
end

class Hello < Greeting #継承している
    # この下に、say_helloメソッドを記述する
    def say_hello()
        puts "#{@msg} #{@target}"
    end
end

player = Hello.new()
player.say_hello()
```
## オーバーライド
```ruby
class Box
    def initialize(item)
        @item = item
    end

    def open()
        puts "宝箱を開いた。#{@item}を手にいれた。"
    end
end

class MagicBox < Box
    def open() # openメソッドのオーバーライド、再定義した
        puts "宝箱を開いた。#{@item}が襲いかかってきた！"
    end
end

box = Box.new("鋼鉄の剣")
box.open()

magicbox = MagicBox.new("オバケ")
magicbox.open()
```

## オーバーライドさせることで出力を変化させる
```ruby
class Player
    def initialize(name)
        @name = name
    end

    def attack(enemy)
        puts  "#{@name}は#{enemy}を攻撃した"
    end
end

class Warrior < Player
    def attack(enemy)
        puts  "#{@name}は#{enemy}を猛攻撃した！"
    end
end


team = []
team.push(Player.new("勇者"))
team.push(Player.new("魔法使い"))
team.push(Warrior.new("戦士")) #<=ここで呼び出すクラスを変更してる
team.each do |person|
    person.attack("スライム")
end
```
## super, privateメソッド
```ruby
# クラスからメソッドを呼び出す

class Player
    def initialize(name)
        @name = name
    end

    def attack(enemy)
        puts "#{@name}は、#{enemy}を攻撃した！"
    end
end

class Wizard < Player
    def initialize()
        super("魔法使い") 　　　　　　　　　　#　<=オーバーライド元のinitializeメソッドをそのまま使いたいからsuperメソッドを使う
    end
    def attack(enemy)
        spell()             #　<=privateメソッドを呼び出している
        puts "#{@name}は、#{enemy}に炎を放った！"
    end
    
    private　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　　 #　<=Wizardクラス以外から呼び出せなくする
    def spell()
        puts "シャラララーン！"
    end
end


puts "=== パーティでスライムと戦う ==="
hero   = Player.new("勇者")
warrior = Player.new("戦士")
wizard = Wizard.new()
party = [hero, warrior, wizard]

party.each do |player|
    player.attack("スライム")
end
```
## クラスメソッドを定義する
```ruby
class Player
    @@charactor_count = 0
    def self.charactor_count()  # selfを使う
        @@charactor_count
    end
end

puts "#{Player.charactor_count()}人で、スライムを攻撃した"

#クラスメソッドでは、メソッド名にselfを付けて定義します。
#クラスメソッドを呼び出す時には、クラス名に続けて、メソッドを記述します。
```

## モジュール

```ruby
モジュールの定義
module Greeting
    def say_hello()
    	puts "hello world"
    end
end
```
```ruby
モジュールのメソッドを呼び出す
module Greeting
    def say_hello()
    	puts "hello world"
    end

    module_function :say_hello
end

Greeting.say_hello()
```
```ruby
クラスにモジュールを組みこむ
module Greeting
    def say_morning()
        puts "good morning! world"
    end
end

class Player
    include Greeting

    def introduce()
        puts "私は、勇者です。"
    end
end

player = Player.new()
player.introduce()
player.say_morning()
```
