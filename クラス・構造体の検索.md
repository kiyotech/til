N = gets.chomp.to_i
array = []

    array << gets.chomp.split

k = gets.to_i

 end

### クラス使った時
```ruby
class Student
attr_accessor :name, :old, :birth, :state
    def initialize(name, old, birth, state)
        @name = name
        @old = old
        @birth = birth
        @state = state
    end
end

n = gets.chomp.to_i

roster = Array.new(n)
n.times do |i|
    name, old, birth, state = gets.chomp.split
    roster[i] = Student.new(name, old, birth, state)
end

k = gets.chomp
roster.each do |student|
    if student.old == k
        puts student.name
        break
    end
end
```
構造体の変更
```ruby
class Student
attr_accessor :name, :old, :birth, :state

    def initialize(name, old, birth, state)
        @name = name
        @old = old
        @birth = birth
        @state = state
    end
    
    def change_name(name)
        @name = name
    end
end

n, k = gets.chomp.split.map(&:to_i)

roster = []
n.times do
    name, old, birth, state = gets.chomp.split
    roster << Student.new(name, old, birth, state)
end

k.times do
    index, new_name = gets.chomp.split
    roster[index.to_i - 1].change_name(new_name)
end

roster.each do |student|
    puts "#{student.name} #{student.old} #{student.birth} #{student.state}"
end
```

## 数の一致しない値を取得
```ruby
class Employee
    def initialize(number, name)
    @number = number
    @name = name
    end
    
    def getnum()
        return @number
    end
    
    def getname()
        return @name
    end
end

n = gets.chomp.to_i
roster = []

n.times do
    values = gets.chomp.split()
    query = values[0]
    if query == "make"
        number = values[1].to_i
        name = values[2]
        roster.push(Employee.new(number, name))
    elsif query == "getnum"
        index = values[1].to_i - 1
        number = roster[index].getnum()
        puts number
    else
        index = values[1].to_i - 1
        name = roster[index].getname()
        puts name
    end
end
```
