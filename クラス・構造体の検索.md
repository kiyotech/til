### クラス使和ない時
```ruby
N = gets.chomp.to_i
array = []
N.times do
    array << gets.chomp.split
end
k = gets.to_i
N.times do |i|
    if array[i][1] == k.to_s
      puts "User{"
      puts "nickname : #{array[i][0]}"
      puts "old : #{array[i][1]}"
      puts "birth : #{array[i][2]}"
      puts "state : #{array[i][3]}"
      puts "}"
    end
 end
        
```
### クラス使った時
```ruby
class Student
attr_accessor :name, :old, :birth, :state
    def initialize(name, old, birth, state)
        @name = name
        @old = old
        @birth = birth
        @state = state
    end
end

n = gets.chomp.to_i

roster = Array.new(n)
n.times do |i|
    name, old, birth, state = gets.chomp.split
    roster[i] = Student.new(name, old, birth, state)
end

k = gets.chomp
roster.each do |student|
    if student.old == k
        puts student.name
        break
    end
end
```
