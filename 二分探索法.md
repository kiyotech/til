## 二分探索法
```
n = gets.to_i
A = gets.split.map(&:to_i)
q = gets.to_i
(0...q).each do
    k = gets.to_i
    left = 0
    right = n-1
    found = false
    while left <= right

        mid = (left + right) / 2
        p mid
        if A[mid] == k 
            puts "Yes"
            found = true
            break
        elsif A[mid] < k 
            left = mid + 1
        else 
            right = mid - 1
        end
    end
    puts "No" unless found
end```
