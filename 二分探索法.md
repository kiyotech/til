## 二分探索法
```
10
3 11 18 25 40 58 69 81 88 99
5
11
100
2
41
69
```
```ruby
n = gets.to_i
A = gets.split.map(&:to_i)
q = gets.to_i
(0...q).each do
    k = gets.to_i
    left = 0
    right = n-1
    found = false
    while left <= right

        mid = (left + right) / 2
     
        if A[mid] == k 
            puts "Yes"
            found = true
            break
        elsif A[mid] < k 
            left = mid + 1
        else 
            right = mid - 1
        end
    end
    puts "No" unless found
end
```
```ruby
def binary_search(a, n, k)
  left = 0
  right = n-1
  while left <= right
    mid = (left + right) / 2

    if a[mid] == k
      return "Yes"
    elsif a[mid] < k
      left = mid + 1
    else
      right = mid - 1
    end
  end

  return "No"
end

n = gets.to_i
A = gets.split.map(&:to_i)
q = gets.to_i

(0...q).each do
  k = gets.to_i
  puts binary_search(A, n, k)
end
```
