### for
```ruby
for i in 0..9
  puts "あ"
end # => あが９回出力される
```
```ruby
for i in 0..15
    puts i
end # => ０から１５まで出力される
```
```ruby
for i in 1..12
    puts "#{i}月"
end # => 1月から１２月まで出力される
```
---
### while
```ruby
# whileによるループ処理
i = 1 # カウンタ変数を初期化
while i <= 10
    puts i　# 繰り返し処理
    i = i + 1　# カウンタ変数を更新
end　　# => 1から１０まで出力
```
